\chapter{Infinitary Logic}

This project formalizes the theory of infinitary logic ($\Lomegaone$), Scott sentences,
back-and-forth equivalence, and Karp's theorem in Lean~4 using Mathlib.

\section{Back-and-Forth Equivalence}

\begin{definition}[Back-and-forth equivalence]\label{def:BFEquiv}\lean{BFEquiv}\leanok
  For structures $M$ and $N$ over a language $L$, ordinal $\alpha$, and tuples
  $a \in M^n$, $b \in N^n$, we define $\BFEquiv_\alpha(a,b)$ by transfinite recursion:
  \begin{itemize}
    \item $\alpha = 0$: $a$ and $b$ satisfy the same atomic formulas.
    \item $\alpha = \beta + 1$: for every $m \in M$ there exists $n' \in N$ with
      $\BFEquiv_\beta(a \cdot m, b \cdot n')$, and symmetrically.
    \item $\alpha$ limit: $\BFEquiv_\beta(a,b)$ for all $\beta < \alpha$.
  \end{itemize}
\end{definition}

\section{Scott Formulas and Sentences}

\begin{definition}[Scott formula]\label{def:scottFormula}\lean{scottFormula}\leanok
  \uses{def:BFEquiv}
  For a tuple $a \in M^n$ and ordinal $\alpha < \omegaone$, the Scott formula
  $\sigma_\alpha(a)$ is an $\Lomegaone$-formula capturing back-and-forth equivalence
  at level $\alpha$.
\end{definition}

\begin{theorem}[Scott formula characterization]\label{thm:scottFormula-iff}
  \lean{realize_scottFormula_iff_BFEquiv}\leanok
  \uses{def:scottFormula, def:BFEquiv}
  For countable $M$, $\sigma_\alpha(a)$ is realized by $b$ in $N$ if and only if
  $\BFEquiv_\alpha(a,b)$.
\end{theorem}

\section{Self-Stabilization}

\begin{theorem}[Self-stabilization]\label{thm:self-stabilization}
  \lean{exists_complete_self_stabilization}\leanok
  \uses{def:BFEquiv}
  For any countable $L$-structure $M$, there exists an ordinal $\alpha < \omegaone$
  such that $M$ self-stabilizes completely at level $\alpha$: for every tuple $a$,
  $\BFEquiv_\alpha(a,a') \iff \BFEquiv_{\alpha+1}(a,a')$ within $M$.
\end{theorem}

\section{Scott Sentence}

\begin{definition}[Stabilization ordinal]\label{def:stabilization-ordinal}
  \lean{stabilizationOrdinal}\leanok
  \uses{def:BFEquiv}
  The stabilization ordinal of a countable structure $M$ is the least ordinal $\alpha$
  such that the Scott analysis stabilizes at level $\alpha$.
\end{definition}

\begin{definition}[Scott sentence]\label{def:scott-sentence}
  \lean{scottSentence}\leanok
  \uses{def:scottFormula, def:stabilization-ordinal}
  The Scott sentence of a countable structure $M$ is the Scott formula for the empty
  tuple at the stabilization ordinal: $\sigma_{\alpha_0}(\langle\rangle)$.
\end{definition}

\begin{theorem}[Scott sentence characterization (conditional)]\label{thm:scott-characterizes-of}
  \lean{scottSentence_characterizes_of}\leanok
  \uses{def:scott-sentence, thm:self-stabilization, thm:scottFormula-iff}
  Assuming the countable refinement hypothesis, the Scott sentence of a countable
  structure $M$ characterizes $M$ up to isomorphism among countable structures.
\end{theorem}

\chapter{Countable Refinement}

\section{Countable Refinement Hypothesis}

\begin{theorem}[Countable Refinement Hypothesis]\label{thm:CRH}
  \lean{countableRefinementHypothesis}\leanok
  \uses{thm:self-stabilization}
  For any countable relational language $L$, countable $L$-structure $M$,
  tuple size $n$, and tuple $a \in M^n$, the set of refinement ordinals
  $\{\varepsilon < \omegaone \mid \exists (N,b)\,
  \BFEquiv_\varepsilon(a,b) \wedge \neg\BFEquiv_{\varepsilon+1}(a,b)\}$
  is countable.
\end{theorem}

This countability result drives the downstream stabilization argument: since
only countably many ordinals can witness a strict refinement, the hierarchy
must stabilize below~$\omegaone$.

\begin{theorem}[Scott sentence characterization]\label{thm:scott-characterizes}
  \lean{scottSentence_characterizes}\leanok
  \uses{thm:CRH, thm:scott-characterizes-of}
  The Scott sentence of a countable structure $M$ characterizes $M$ up to isomorphism
  among countable structures. (Unconditional corollary of the CRH.)
\end{theorem}

\section{Scott Rank and Height}

\begin{definition}[Scott rank]\label{def:scottRank}\lean{scottRank}\leanok
  \uses{def:BFEquiv}
  The Scott rank of a countable structure $M$ is
  $\scottRank(M) = \sup_{m \in M}(\mathrm{er}(m) + 1)$, where $\mathrm{er}(m)$
  is the element rank.
\end{definition}

\begin{theorem}[Scott rank below $\omegaone$]\label{thm:scottRank-lt-omega1}
  \lean{scottRank_lt_omega1}\leanok
  \uses{def:scottRank, thm:CRH}
  For any countable $L$-structure $M$, $\scottRank(M) < \omegaone$.
\end{theorem}

\begin{definition}[Scott height]\label{def:scottHeight}\lean{scottHeight}\leanok
  \uses{def:BFEquiv}
  The Scott height of a countable structure $M$ is the least ordinal $\alpha$ such
  that $\BFEquiv_\alpha(a,b)$ implies $\BFEquiv_{\alpha+1}(a,b)$ for all tuples $a,b$
  in any countable structures.
\end{definition}

\begin{theorem}[Scott height below $\omegaone$]\label{thm:scottHeight-lt-omega1}
  \lean{scottHeight_lt_omega1}\leanok
  \uses{def:scottHeight, thm:CRH}
  For any countable $L$-structure $M$, $\scottHeight(M) < \omegaone$.
\end{theorem}

\chapter{Karp's Theorem}

\begin{definition}[Potential isomorphism]\label{def:potential-iso}
  \lean{PotentialIso}\leanok
  \uses{def:BFEquiv}
  A potential isomorphism between structures $M$ and $N$ is a family of finite partial
  maps containing the empty map and closed under extension in both directions.
\end{definition}

\begin{definition}[$L_{\infty\omega}$-equivalence]\label{def:linf-equiv}
  \lean{LinfEquivW}\leanok
  $L_{\infty\omega}$-equivalence between structures $M$ and $N$: they satisfy the same
  sentences of $L_{\infty\omega}$.
\end{definition}

\begin{theorem}[Karp's theorem]\label{thm:karp-theorem}
  \lean{karp_theorem_w}\leanok
  \uses{def:potential-iso, def:linf-equiv, def:BFEquiv}
  Two structures $M$ and $N$ are back-and-forth equivalent (there exists a potential
  isomorphism) if and only if they are $L_{\infty\omega}^w$-equivalent (satisfy the same
  $L_{\infty\omega}$ sentences with index types in universe $w$).
\end{theorem}
